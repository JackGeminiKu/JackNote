常见操作
获取长度

length属性

访问数组元素

scores[0]

添加/删除元素

scores[2] = 100; // 如果下标为2的元素存在, 则将它的值改为100; 否则, 添加一个下标为2的元素, 它的值为100.
array.push(...); // 将参数添加到原数组末尾，并返回数组的长度 .
array.ushift(...); // 将参数添加到原数组开头，并返回数组的长度

array.pop(...);  // 删除数组最后一项，并返回删除元素的值；如果数组为空则返回undefined
array.shift(...); // 删除数组第一项，并返回删除元素的值；如果数组为空则返回undefined
splice

// splice(start,deleteCount,val1,val2,...)：从start位置开始删除deleteCount项，并从该位置起插入val1,val2,... 返回的是含有被删除的元素的数组.
var a = [1,2,3,4,5];
var b = a.splice(2,2,7,8,9); //a：[1,2,7,8,9,5] b：[3,4]
var b = a.splice(0,1); //同shift
a.splice(0,0,-2,-1); var b = a.length; //同unshift
var b = a.splice(a.length-1,1); //同pop
a.splice(a.length,0,6,7); var b = a.length; //同push
join

// join(separator)：将数组的元素组起一个字符串，以separator为分隔符，省略的话则用默认用逗号为分隔符
var a = [1,2,3,4,5];
var b = a.join("|"); //a：[1,2,3,4,5] b："1|2|3|4|5"
sort

// sort(orderfunction)：按指定的参数对数组进行排序
var a = [1,2,3,4,5];
var b = a.sort(); //a：[1,2,3,4,5] b：[1,2,3,4,5]
concat

// concat：返回一个新数组，是将参数添加到原数组中构成的
var a = [1,2,3,4,5];
var b = a.concat(6,7); //a：[1,2,3,4,5] b：[1,2,3,4,5,6,7]
reverse

// reverse：将数组反序
var a = [1,2,3,4,5];
var b = a.reverse(); //a：[5,4,3,2,1] b：[5,4,3,2,1]
slice

// slice(start,end)：返回从原数组中指定开始下标到结束下标之间的项组成的新数组
var a = [1,2,3,4,5];
var b = a.slice(2,5); //a：[1,2,3,4,5] b：[3,4,5];
