创建对象

使用构造函数定义


function Dog(name, breed, weight) {           // 构造函数首字母大写! 没有return哦!!!
    this.name = name;
    this.breed = breed;
    this.weight = weight;
    this.bark = function() {
        alert(this.name + " says: Woof!");
    };
}

// 给Dog类添加静态方法
Dog.staticMethod = function() {
    alert("staticMethod");
}

// 添加实例方法
Dog.prototype.instanceMethod = function() {
    alert("instanceMethod");
}

// 构造对象
var fido = new Dog("Fido", "Mixed", 40);

字面量对象: 用大括号和逗号分隔的属性所写的对象.


var fido = {
    name: "Fido",                             // 属性
    weight: 40,
    breed: "Mixed",
    loves: ["walks", "fetching balls"],
    bark: function() {                        // 方法
        // this: 表示调用这个方法的对象
        alert(this.name + ": Woof woof!");    
    }
};

使用属性

// 使用.运算符(推荐这种)
fido.weight = 25;
alert(fido.weight);    // 访问不存在的属性, 会得到underfined值.

// 使用[]运算符
fido["weight"] = 25;
alert(fido["weight"]);
调用方法

// 调用静态方法
Dog.staticMethod();
// 调用实例方法

增/删对象属性

fido.age = 5;    // 为对象增加属性

delete fido.age;    // 如果删除属性成功(或者删除了一个不存在的属性), delete表达式会返回true
遍历对象属性

var prop;
for (prop in fido) {    // prop是fido的属性名对应的字符串值
    alert("我是属性: " + prop + ", 属性值是: " + fido[prop]);
}
扩展方法

function addhead (level) {
    html = “H” + level;
    text = this.toString();
    start = “<” + html + “>”;
    stop = “</” + html + “>”;
    return start + text + stop;
}
String.prototype.heading = addhead;

document.write (“This is a test”.heading(1));
